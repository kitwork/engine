app.js
const { router, log, database, go } = kitwork();

const db = database({ type: "postgres" });

router.get("/users/:id?").handle((req, res) => {
    // 1. Database query via ORM
    const users = db.user.limit(5).list();

    // 2. Spin up parallel Goroutine
    go(() => {
        log.Print("⚡ Isolated background sync");
    });

    return res.json({ data: users });
});
